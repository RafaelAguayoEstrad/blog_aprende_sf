Java

Java es un lenguaje de programación orientado a objetos, robusto y ampliamente utilizado en el desarrollo de una variedad de aplicaciones, incluyendo aplicaciones empresariales, aplicaciones móviles, sistemas embebidos y servicios web.

Algunos conceptos importantes de Java incluyen:


  • Clase: es un prototipo o plantilla, que define un conjunto de atributos que conforman las caracteristicas para poder crear un objeto de esta.

    Una clase es un conjunto de objetos similares que comparten caracteristicas y comportamientos comunes.
    Objeto: entidad compuesta de datos y de operaciones que se realizan sobre esos datos. Es una entidad provista de un conjunto de atributos y de comportamientos. Un objeto es una instancia (un ejemplar un caso concreto) de una clase (una categoria) y es una representacion concreta especifica de la clase. Atributos: son las caracteristicas propias de un objeto.


  • Metodo: acciones o compartamientos de un objeto.

  • ARREGLOS MULTIDIMENCIONALES

    • Java no soporta arreglos bidimensionales, pero se puede lograr la misma finalidad con un arreglo de arreglos con el tamaño de los arreglos definidos.

  • Ejemplo 1:
    • 
    
    
int x[][] = new int[3][2];

Agregar elementos:
x[0][1] = 10 , x[2][1] = 20



// Ejemplo 1
public class Ejemplo1 {
    public static void main(String[] args) {
        int[][] x = new int[3][2];
        // Agregar elementos
        x[0][1] = 10;
        x[2][1] = 20;

        // Imprimir la matriz
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 2; j++) {
                System.out.print(x[i][j] + " ");
            }
            System.out.println();
        }
    }
}

    void plantarArbol();

class Arbol {
    float altura;

    public Arbol(float altura) {
        this.altura = altura;
    }
}

class Encino extends Arbol implements ArbolGrande, ArbolAccion {
    public Encino(float altura) {
        super(altura);
    }

    public void asignaAltura(float altura) {
        this.altura = altura;
        JOptionPane.showMessageDialog(null, "Se asignó la altura del encino a: " + altura);
    }

    public float obtieneAltura() {
        JOptionPane.showMessageDialog(null, "La altura del encino es: " + altura);
        return altura;
    }

    public void cortarArbol() {
        JOptionPane.showMessageDialog(null, "Se cortó el encino");
    }

    public void plantarArbol() {
        JOptionPane.showMessageDialog(null, "Se plantó el encino");
    }

    public static void main(String[] args) {
        Encino encino = new Encino(0.0f);

        encino.cortarArbol();
        encino.plantarArbol();
        encino.asignaAltura(0.0f);
        encino.obtieneAltura();
    }

import javax.swing.JOptionPane;
import java.util.Stack;

class Pila1 {
    public static void main(String[] args) {
        int tam = Integer.parseInt(JOptionPane.showInputDialog("¿Cuántos elementos quieres en el arreglo?"));
        int[] arreglo = new int[tam];

        // Llenar el arreglo
        for (int i = 0; i < tam; i++) {
            arreglo[i] = Integer.parseInt(JOptionPane.showInputDialog("Ingrese un número entero para la posición " + i));
        }

        // Crear la pila
        Stack pila = new Stack();

        // Llenar la pila con los elementos del arreglo
        for (int i = 0; i < tam; i++) {
            pila.push(arreglo[i]);
        }

        // Mostrar la pila en pantalla
        System.out.println("Elementos de la pila en orden inverso:");
        for (int i = 0; i < tam; i++) {
            System.out.println("Elemento: " + pila.pop());
        }
    }
}

import javax.swing.JOptionPane;
import java.util.Queue;
import java.util.LinkedList;

class Aplica1 {
    public static void main(String args[]) {
        Queue q = new LinkedList();

        String numElementosString = JOptionPane.showInputDialog("Ingrese el número de elementos:");
        int numElementos = Integer.parseInt(numElementosString);

        for (int i = 0; i < numElementos; i++) {
            String elementoString = JOptionPane.showInputDialog("Ingrese el elemento " + (i + 1) + ":");
            int elemento = Integer.parseInt(elementoString);
            q.add(elemento);
        }

        int suma = 0;
        while (!q.isEmpty()) {
            suma += q.poll();
        }

        JOptionPane.showMessageDialog(null, "Suma de los elementos: " + suma);
    }
}
